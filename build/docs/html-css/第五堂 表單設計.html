  1. 手寫 SCSS 結構參考

    • 手刻

      @import "variable";// 變數
      @import "reset";  // reset.css

      // 1. 元件(進階)*
      @import "button";
      @import "...";

      // 2. 雜項(進階)*
      @import "utilities";

      // 3. 共用樣式（如有 Bootstrap 通常不會寫這段）
      @import "layout"; // 共同框架、第一層

      // 4. 分頁樣式（可在此覆蓋 Bootstrap 樣式）
      @import "index";
      @import "page1";

    • Bootstrap

      @import "variable"; // 變數
      @import "bootstrap";

      // 1. 自定義元件(進階)*
      @import "button";
      @import "...";

      // 2. 雜項(進階)*
      @import "utilities";

      // 3. 分頁樣式（可在此覆蓋 Bootstrap 樣式）
      @import "index";
      @import "page1";
